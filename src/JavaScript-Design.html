Search Record in Table

function tablesearch(){
    let input  = document.getElementById('myinput').value; //getting input value
    let filter = input.toUpperCase();   // convert search value to toUpperCase
    let a = document.getElementById('myTable'); //get table
    let tr = a.getElementsByTagName('tr');  //get table rows

    for (let i = 1; i < tr.length; i++) {
        let td = tr[i].getElementsByTagName('td')[0];

        if (td) {
            let text = td.textcontent || td.innerHTML;
            if (text.toUpperCase().indexOf(filter) > -1) {
                tr[i].style.display = "";
            } else {
                tr[i].style.display = "none";
            }
        }
    }
}

        function openCity(e, cityName) {
            var i, tabcontent, tablink;

            //display :none to the tabcontent b/c they not show on click
            tabcontent = document.getElementsByClassName('tabcontent');
            for (i = 0; i < tabcontent.length; i++) {
                tabcontent[i].style.display = "none";
            }

            //use for the active button not click
            tablink = document.getElementsByClassName('tablink');
            for (i = 0; i < tablink.length; i++) {
                tablink[i].className = tablink[i].className.replace('active');
            }

            document.getElementById(cityName).style.display = "block";
            e.currentTarget.className += " active";
        }
