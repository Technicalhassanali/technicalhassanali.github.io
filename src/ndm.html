Network Design & Management

Create VLAN in Switch For DHCP Step 01 

switch># en(enable)
switch># config terminal
                    // First Vlan Create
switch(config)# vlan 10
switch(config-vlan)# name Hassan
switch(config-vlan)# exit
					// Second Vlan Create
switch(config)# vlan 20
switch(config-vlan)# name Ali
switch(config-vlan)# exit
                   // Third Vlan Create
switch(config)# vlan 30
switch(config-vlan)# name Khan
switch(config-vlan)# exit

Assigning the VLAN To Pc Step 02 

switch(config)# interface fastEthernet0/1
switch(config-if)# interface fastEthernet0/1
switch(config-if)# interface fastEthernet0/1
switch(config-if)# switchport mode access
switch(config-if)# switchport access vlan 10
switch(config)# exit
/* Same Method Apply for other vlan*/

List of VLAN Created In Switch 

Method 01 : switch(config)# do show vlan brief
Method 02 : switch# show vlan brief

Made Switchport Trunk

	switch(config)# interface fastEthernet0/1
	switch(config-if)# no shutdown
	switch(config-if)# switchport mode trunk

Config Router As DHCP Step 03

Select RouterISR4321 insert the port in gigathernet.

		Router# ip dhcp pool ?            // list of pool
	Router# ip dhcp binding ?        // list of binding hardware
	/*Creating Pool */
	Router(config)# interface gi0/0/0
	Router(config-if)# no shut
	Router(config-if)# exit
	Router(config)# exit
	Router# show interface gi0/0/0
	// Creating Sub Interface of One Port
	// Creating 1st Interface
	Router(config)# interface gi0/0/0.1
	Router(config-subif)# interface gi0/0/0.1
	Router(config-subif)# ip address 10.0.0.1 255.255.255.0
	Router(config-subif)# encapsulation dot1Q ?
	Router(config-subif)# encapsulation dot1Q 10
	Router(config-subif)# exit

	Router(config)# interface gi0/0/0.2
	Router(config-subif)# ip address 20.0.0.1 255.255.255.0
	Router(config-subif)# encapsulation dot1Q ?
	Router(config-subif)# encapsulation dot1Q 20


	Router(config)# interface gi0/0/0.3
	Router(config-subif)# ip address 30.0.0.1 255.255.255.0
	Router(config-subif)# encapsulation dot1Q ?
	Router(config-subif)# encapsulation dot1Q 30
	Router(config-subif)# exit


	Router(config)# ip dhcp pool dhcpvlan10(poolname)
	Router(config)# network 10.0.0.0 255.255.255.0
	Router(config)# default-router 10.0.0.1 (Gateway)
	
	Router(config)# ip dhcp pool dhcpvlan20(poolname)
	Router(config)# network 20.0.0.0 255.255.255.0
	Router(config)# default-router 20.0.0.1 (Gateway)

	Router(config)# ip dhcp pool dhcpvlan30(poolname)
	Router(config)# network 30.0.0.0 255.255.255.0
	Router(config)# default-router 30.0.0.1 (Gateway)

	Router# ip dhcp pool ?            // list of pool
	Router# ip dhcp binding ?        // list of binding hardware
